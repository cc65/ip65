IP65

IP65 is a TCP/IP stack for 6502 based computers.

Sample UDP listener source

gangedport      = 60064

        jsr ip65_init
        lda #<gotpacket
        ldx #>gotpacket
        sta udp_callback
        stx udp_callback + 1
        lda #<gangedport
        ldx #>gangedport
        jsr udp_add_listener

main:
        jsr ip65_process
        jmp main

gotpacket:
        sei
        lda $01
        pha
        lda udp_inp
        sta $01

        lda udp_inp + 1
        ldx udp_inp + 2
        sta zp_data
        stx zp_data + 2
        ldy udp_inp + 3
copy:
        lda udp_inp + 3,y
        sta (zp_data),y
        dey
        bne copy

        pla
        sta $01
        cli
        rts
